# Mtodo create_app que recebe como argumento com todas as configuraes da aplicao.
def create_app(config_name):

    # A varivel app recebe uma instncia de Flask, passando a configurao da localizao dos templates. Pode 
    # receber diversas configuraes.
    app = Flask(__name__, template_folder='templates')

    # O atributo secret_key da aplicao app, recebe a configurao da chave secreta do arquivo config.py, por meio da varivel config e atributo SECRET.
    app.secret_key = config.SECRET 

    # Efetua o carregamento do arquivo config.py
    app.config.from_object(app_config[config_name]) 
    app.config.from_pyfile('config.py')

    # Adicionado para migrar o banco de dados
    app.config['SQLALCHEMY_DATABASE_URI'] = config.SQLALCHEMY_DATABASE_URI

    # Configurao indicando que as modificaes do banco de dados podero ser exclusivamente pela aplicao.
    # Se True, permite outras formas.
    app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = config.SQLALCHEMY_TRACK_MODIFICATIONS

    # A varivel db receber a atribuio de uma instncia do SQLAlchemy passando a aplicao app criada.
    db = SQLAlchemy(config.APP)

    # Adicionado a inicializao das visualizaes das views Admin. Ativa a rea admin do Flask.
    # A Admin ser criada dentro do arquivo admin/Admin.py
    start_views(app,db)                                             <--- Adicione

    # Inicializando o banco de dados para a aplicao.
    db.init_app(app)


# <--- Adequar daqui pra frente ------->
    # Primeira rota a ser criada. A rota principal root.
    # Quando se cria duas ou mais rotas seguidas antes do mtodo, isso indica ao navegador que qualquer uma delas acessa o mtodo a seguir. Neste exemplo tanto http://localhost:8000/ e http://localhost:8000/login/ acessam o mdtodo index(). Que retorna uma string.

    def login_post():

        return '<h1>Aqui entrar a tela de recuperar senha. </h1>'

    @app.route('/recuperar-senha/', methods=['POST']) 
    def enviar_recuperar_senha():
        # Instncia do Usuario Controller atribuda a varivel usuario.
        usuario = UsuarioController()
        resultado = usuario.recuperar(request.form['email'])

        if resultado:

        else:

            return render_template('recuperar.html', data={'status': 401, 'msg': 'Erro ao enviar e-mail de recuperao'})

    return app

# -*- coding: utf-8 -*-
# from flask_admin import Admin: Ela possui vrios recursos para criao 
# de views customizadas para nossa aplicao
from flask_admin import Admin

# Componente dentro do flask_admin para ativar as views em nossa aplicao, baseadas em nossas models.
from flask_admin.contrib.sqla import ModelView
from models.Perfil import Perfil
from models.Usuario import Usuario
from models.Area import Area 
from models.Modulo import Modulo
from models.Componente import Componente


def start_views(app, db):
    # Utilizao do bootstrap4 por meio do construtor do Admin.
    admin = Admin(app, name='Componentes Curriculares', template_mode='bootstrap4')
    # Mtodo que  utilizado para criar uma view em nossa aplicao Admin.
    # ModelView:  um recurso do flask_admin que permite a criao Admin, as telas do administrador baseadas na estrutura das models.
    # category  utilizado para agrupar os itens de menus.
    admin.add_view(ModelView(Perfil, db.session, "Perfis Acesso", category= "Configuraes"))
    admin.add_view(ModelView(Usuario, db.session, "Usurios", category="Usurios"))
    admin.add_view(ModelView(Area, db.session, "reas", category="Configuraes"))
    admin.add_view(ModelView(Modulo, db.session, "Mdulos", category="Configuraes"))
    admin.add_view(ModelView(Componente, db.session, "Componentes", category="Componentes"))
